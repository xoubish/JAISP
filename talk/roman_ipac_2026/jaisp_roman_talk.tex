\begin{frame}{Why combine surveys at the pixel level?}
  \begin{itemize}\setlength\itemsep{0.7em}
    \item Cosmology is now limited by \alert{systematics, not statistics}: the remaining
      disagreements between cosmological probes are too significant to ignore, but we cannot
      yet tell measurement error apart from new physics.
    \item Rubin/LSST, \textit{Euclid}, and \textit{Roman} will observe overlapping sky from
      ${\sim}0.3$ to $2\,\mu$m at sub-arcsecond resolution --- in combination they can
      \alert{calibrate and cross-check one another}, suppressing survey-specific systematics
      no single survey can control alone.
    \item Working at the pixel level means handling two instruments with \alert{different
      resolutions, PSFs, depths, and noise} at the same time --- something no
      single-survey pipeline was built to do.
    \item Classical pipelines encode decades of problem-specific effort. The goal is not to
      discard them, but to move the \emph{joint, cross-instrument} part of the problem into a
      representation \alert{learned from the pixels themselves}.
  \end{itemize}
\end{frame}

% ------------------------------------------------------------ 3
\begin{frame}{Train one model once --- reuse it for every measurement}
  \begin{columns}[T]
    \begin{column}{0.58\textwidth}
      \textbf{The model trains itself:}
      \begin{itemize}
        \item hide one of the ten bands; predict that band's image from the other nine
        \item the images are their own supervision --- \alert{no labels, no catalogs, no simulations}
      \end{itemize}
      \vspace{0.5em}
      \textbf{Then lock it and reuse it:}
      \begin{itemize}
        \item every measurement afterwards is a small add-on network reading the same
          stored features, at a tiny fraction of the training cost
        \item detection \& astrometry working today; photometry \& shapes in preparation
      \end{itemize}
    \end{column}
    \begin{column}{0.38\textwidth}
      \begin{beamercolorbox}[sep=0.8em,rounded=true]{block body}
        \textbf{Cost structure}\\[0.4em]
        Pretraining: \alert{${\sim}2.5$ GPU-hours}\\[0.2em]
        Model size: ${\sim}9$M parameters\\[0.2em]
        Features: computed once per dataset\\[0.2em]
        Each new task: a small add-on network\\[0.4em]
        {\small $\Rightarrow$ marginal cost per task $\to$ small; per field $\to$ zero (as we will see)}
      \end{beamercolorbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Architecture: each instrument at its native resolution}
  \centering
  \includegraphics[width=0.80\textwidth]{fig2_architecture.png}
  \begin{itemize}\small
    \item Rubin stays at $0.2''$/px, \textit{Euclid} at $0.1''$/px --- \alert{no resampling onto a
      shared grid}, so neither instrument's spatial information is thrown away.
    \item The branches merge on a common $0.4''$ grid, where all ten bands exchange information.
  \end{itemize}

% ------------------------------------------------------------ 5b
\begin{frame}{We didn't start here: ten iterations to this design}
  \small
  \begin{tabular}{@{}p{0.09\textwidth}p{0.60\textwidth}l@{}}
    \toprule
    v1--v2 & \textbf{Contrastive patch matching} --- make matching Rubin/\textit{Euclid} cutouts
             look similar in feature space & \textcolor{jaispaccent}{failed: sky-dominated} \\[0.35em]
    v3     & \textbf{Object-centric matching} --- discover individual sources, pair them across
             instruments & \textcolor{jaispaccent}{abandoned: no precision} \\[0.35em]
    v4--v5 & \textbf{Feature matching on full tiles (JEPA)} --- make the two instruments'
             features agree at corresponding positions & \textcolor{jaispaccent}{beaten by a simple CNN} \\[0.35em]
    v6     & \textbf{The turn: predict actual pixels} --- hide one band, reconstruct its image;
             no shortcut that discards position & \alert{\textbf{paradigm shift}} \\[0.35em]
    v7--v8 & \textbf{Mixed resolution} --- one branch per instrument at its native pixel scale,
             merged at a common physical scale & \textcolor{jaispteal}{works} \\[0.35em]
    v9--v10 & \textbf{Refinements} --- every band gets an equal vote in training; the loss
             focuses on real sources & \textcolor{jaispteal}{\textbf{production}} \\
    \bottomrule
  \end{tabular}
  \vfill
  \centering
  \emph{Features that look similar aren't features that know \textbf{where} ---
    the year of negative results is the design.}
\end{frame}

% ------------------------------------------------------------ 8
\begin{frame}{Downstream task 1 --- Detection}
  \centering
  \includegraphics[height=0.60\textheight]{fig5_detection.png}
  \begin{itemize}\small
    \item A small detection network turns the ten-band features into a \alert{source-evidence
      map} --- finding sources is still peak-finding, as in classical detection, but on all
      ten bands at once. Trained on lightweight classical labels ($3\sigma$ VIS extraction).
    \item Matches the \textit{Euclid} Q1 MER catalog on held-out sky; returns
      \alert{one detection per galaxy} where classical extraction fragments extended systems.
  \end{itemize}
\end{frame}

% ------------------------------------------------------------ 9
\begin{frame}{Detection: ten bands buy real depth}
  \begin{columns}[T]
    \begin{column}{0.60\textwidth}
      \centering
      \includegraphics[width=\textwidth]{fig6_detection_performance.png}
    \end{column}
    \begin{column}{0.40\textwidth}
      \vspace{0.5em}
      \begin{itemize}\small\setlength\itemsep{0.6em}
        \item Depth measured by hiding real sources, dimmed to a chosen magnitude, in the
          images and counting how many are found: $50\%$ recovery at \alert{VIS $\approx 25.0$},
          \alert{$+0.45$ mag deeper} than the same detector on VIS alone.
        \item Against the \textit{Euclid} catalog: \alert{93\% completeness, 95\% purity}
          (an undercount --- some ``false'' detections are real sources the catalog missed).
      \end{itemize}
      \vspace{0.3em}
      {\small
      \begin{tabular}{lcc}
        \toprule
        50\% depth (VIS) & ten-band & per-band \\
        \midrule
        all 10 bands & $25.0$ & $25.1$ \\
        VIS only     & $24.6$ & $25.1$ \\
        \alert{multi-band gain} & \alert{$+0.45$} & $+0.03$ \\
        \bottomrule
      \end{tabular}}\\[0.4em]
      {\small An identical detector reading per-band features gains almost nothing --- the
      depth comes from the \alert{combined ten-band representation} itself.}
    \end{column}
  \end{columns}
\end{frame}

% ------------------------------------------------------------ 10
\begin{frame}{Downstream task 2 --- Astrometry: carrying VIS precision to every band}
  \centering
  \includegraphics[width=0.76\textwidth]{fig8_head_correction.png}
  \begin{itemize}\small
    \item The same sources measured in both surveys disagree by ${\sim}50\mas$ --- mostly
      photon noise in centroiding faint galaxies. A small correction network reads the model's
      features at each source and shrinks this to \alert{$14$--$17\mas$ in every band}.
    \item Is it real? We hide sources at \alert{known positions}: corrected positions err by
      only $19\mas$ at $\mathrm{S/N}=5$ (classical: ${\sim}50\mas$) --- nearly the ${\sim}17\mas$
      precision of the sharp VIS channel itself, now carried into every band.
  \end{itemize}
\end{frame}

% ------------------------------------------------------------ 11
\begin{frame}{The concordance field: how well do two Gaia-tied surveys actually agree?}
  \centering
  \includegraphics[height=0.60\textheight]{fig10_concordance_field.png}
  \begin{itemize}\small
    \item Beneath the per-source scatter sits a coherent \alert{$9$--$10\mas$ pattern} between
      the two surveys' astrometric solutions --- even though both are tied to Gaia; ${\sim}7\mas$
      is shared by \alert{all ten bands}. Two entirely different fitting methods recover the same
      pattern, and the per-source correction absorbs it (${\to}1.5\mas$).
    \item A few-mas map of \alert{how well two independently calibrated surveys actually
      agree} --- quality assurance no single pipeline can produce.
  \end{itemize}
\end{frame}

% ------------------------------------------------------------ 12
\begin{frame}{Does any of this transfer? A second deep field, nothing retrained}
  \centering
  \includegraphics[width=0.88\textwidth]{fig11_ood.png}
  \begin{itemize}\small
    \item The locked model and its measurement networks, applied to EDF-S: \alert{nothing retrained}.
    \item Detection: $93.4\%/93.1\%$ completeness/purity vs.\ $93.3\%/94.5\%$ on the training
      field. Positions land on the same ${\sim}12$--$18\mas$ floor, within $1$--$3\mas$.
    \item Scaling to more area is a matter of \alert{applying the stack, not rebuilding it}.
  \end{itemize}
\end{frame}

% ------------------------------------------------------------ 13
\begin{frame}{\textit{Roman}: the natural third stream}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Why the model is ready:}
      \begin{itemize}\small\setlength\itemsep{0.4em}
        \item one encoder branch per instrument \emph{is} the design ---
          adding \textit{Roman} means \alert{adding a third branch}, not redesigning the model
        \item training needs only \alert{overlapping public images} --- no labels, no
          simulations --- and hours of GPU time, once WFI imaging overlaps Rubin/\textit{Euclid} sky
      \end{itemize}
      \vspace{0.4em}
      \textbf{What \textit{Roman} adds:}
      \begin{itemize}\small\setlength\itemsep{0.4em}
        \item \alert{sharp, deep near-infrared imaging at $0.11''$/px} --- the infrared
          counterpart of the sharp VIS reference, where today we must rely on resampled NISP
        \item a wide survey of thousands of deg$^2$ \alert{inside the LSST footprint},
          plus deep time-domain fields
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \textbf{What \textit{Roman} gains:}
      \begin{itemize}\small\setlength\itemsep{0.4em}
        \item detection that uses all surveys at once --- for Rubin$+$\textit{Euclid}
          that was worth $+0.45$ mag
        \item \textit{Roman}-quality positions carried into the ground-based bands
        \item \alert{three-way agreement maps}: Rubin--\textit{Euclid}--\textit{Roman}
          astrometry compared at the few-mas level --- quality assurance no
          single pipeline can produce
        \item cross-instrument alignment solved \emph{inside} the model instead of inherited
          from preprocessing (the ${\sim}3\mas$ NISP resampling error we found is exactly that)
      \end{itemize}
      \vspace{0.4em}
      \centering
      \alert{\small Launch window opens August 30 --- the recipe is ready.}
    \end{column}
  \end{columns}
\end{frame}

% ------------------------------------------------------------ 14
\begin{frame}{Takeaways}
  \begin{itemize}\setlength\itemsep{0.9em}
    \item A small (${\sim}9$M-parameter) foundation model, trained \alert{without any labels}
      on public Rubin\,$+$\,\textit{Euclid} images in ${\sim}2.5$ GPU-hours, learns a shared
      ten-band representation precise enough to measure with.
    \item \textbf{Detection}: matches the \textit{Euclid} catalog (93\%/95\%), and ten-band fusion
      pushes \alert{$0.45$ mag deeper} than VIS alone.
    \item \textbf{Astrometry}: VIS-quality positions in every band --- ${\sim}50 \to 14$--$17\mas$,
      verified at $19\mas$ on hidden test sources --- and a few-mas \alert{map of the
      disagreement} between the two surveys' Gaia-tied solutions.
    \item The whole stack \alert{transfers unchanged} to a second deep field. All of this on the
      \emph{earliest} data both surveys have released --- a floor, not a ceiling.
    \item \textit{Roman} slots in as a \alert{third encoder stream}: sharp NIR anchor, three-way
      cross-survey calibration, and one representation reused across tasks, fields ---
      and instruments.
  \end{itemize}
